 2function [] = Plot_Uz(transdutor)
 3[xs,zs] = meshgrid(-0.05:transdutor.disc:0.05,0:transdutor.disc:0.1);
 4% zs = 0:1e-4:.3;
 5% n = length(zs);
 6s = size(xs);
 7U = zeros(s)+zeros(s)*i;
 8p = zeros(s)+zeros(s)*i;
 9
10for n = 1:s(1)
11    for j = 1:s(2)
12        U(n,j) = Calc_Gorkov(xs(n,j),0,zs(n,j),transdutor);
13        p(n,j) = Calc_pressao(xs(n,j),0,zs(n,j),transdutor);
14    end
15end
16
17figure(3)
18surf(xs,zs, U);
19colormap(jet)
20shading interp
21view(2)
22caxis([-10e-11 0])
23axis equal
24
25
26figure(2)
27surf(xs,zs,abs(p))
28colormap(jet)
29shading interp
30view(2)
31caxis([0 2000])
32axis equal
