1%% FUNÇÕES
 2function p_total = Calc_pressao(x, y, z, transdutor)
 3    x_t = transdutor.posx;
 4    y_t = transdutor.posy;
 5    z_t = transdutor.posz;
 6    A = transdutor.A;
 7    normal_v = transdutor.normal;
 8    omega = transdutor.omega;
 9    a = transdutor.a;
10    k = transdutor.k;
11    phase_v = transdutor.phase;
12    n1 = length(x_t);
13    s = size(x);
14    r = zeros([s,n1])+0j;
15    theta = zeros([s,n1])+0j;
16    Pax = zeros([s,n1])+0j;
17    v = zeros([s,n1])+0j;
18    H = zeros([s,n1])+0j;
19    p = zeros([s,n1])+0j;
20    t1 = 0;
21    for i = 1:n1
22
23        r(:,:,i) = calc_r(x_t(i), y_t(i), z_t(i), x, y,z);
24        theta(:,:,i) = calc_th(x_t(i), y_t(i), z_t(i), x, y, z, normal_v, r(:,:,i));
25        Pax(:,:,i) = calc_Pax(A, r(:,:,i));
26        v(:,:,i) = calc_v(k, a, theta(:,:,i));
27        H(:,:,i) = calc_H(v(:,:,i));
28        p(:,:,i) = calc_p(Pax(:,:,i), H(:,:,i), omega, k, t1, r(:,:,i), phase_v(i));
29    end
30    p_total = sum(p,3);
31end
